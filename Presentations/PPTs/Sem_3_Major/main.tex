    \small

    \begin{itemize}

        \item Plasma is overdense if $\omega>\omega_p$.
        \item High harmonics are generated by interaction of laser with overdense plasma.

\begin{frame}
    \frametitle{Effect of Laser Envelope}
    \tiny
    \begin{minipage}[t]{0.35\linewidth}
        1. Sine Sqaured
        \begin{equation*}\label{sin-sq-env}
            P(t)=
            \begin{cases}
                 & \sin^2(\pi t/T) \text{ for } 0 \leq t \le T \\
                 & 0         \;      \text{ otherwise }
            \end{cases}
        \end{equation*}
        2. Gaussian
        \begin{equation*}\label{gaussian-env}
            P(t)=
            \begin{cases}
                 & e^{\frac{-(t-T/2)^2}{2(0.2T)^2}} \text{ for } 0 \leq t \le T \\
                 & 0         \;      \text{ otherwise }
            \end{cases}
        \end{equation*}
        3. Triangular
        \begin{equation*}\label{triangle-env}
            P(t)= 2\times
            \begin{cases}
                 & t/T \text{ for } 0 \leq t \le T/2    \\
                 & 1-t/T \text{ for } T/2 \leq t \le T  \\
                 & 0         \;      \text{ otherwise }
            \end{cases}
        \end{equation*}
    \end{minipage}
